Table perfume {
  perfume_id     int           [pk]               // 향수_ID
  url            varchar(500)                     // 상세_URL
  Perfume        varchar(100)                     // 향수명
  Brand          varchar(50)                      // 브랜드명
  Country        varchar(50)
  Gender         varchar(3)                      // 성별 (women/men/unisex 등, 길이는 여유있게)
  RatingValue   float(3,2)                       // 평균평점
  RatingCount   int                              // 평점개수
  Year     int
  Top      text                             // 탑노트
  Middle   text                             // 미들노트
  Base     text                             // 베이스노트
  mainaccord1   varchar(20)                      // 메인어코드1
  mainaccord2   varchar(20)                      // 메인어코드2
  mainaccord3   varchar(20)                      // 메인어코드3
